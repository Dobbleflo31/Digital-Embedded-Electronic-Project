109 static void GAMEOFLIFE_next_generation(grid_t * grid_prev, grid_t * grid_next)
110 {
111  uint8_t x,y,i,j;
112  for(x=0; x<NB_COLUMN; x++){ //Parcours colonnes
113  for(y=0; y<NB_LINE; y++){ //Parcours lignes
114  uint8_t cell_alive_count = 0;
115  if(x-1 > 0 && x+1 < NB_COLUMN){ //Evite de out of range dans la grille
116  for(i=x-1; i<=x+1; i++){ //Parcours voisins colonnes
117  if(y-1 > 0 && y+1 < NB_LINE){ //Evite de out of range dans la grille
118  for(j=y-1; j<=y+1; j++){ //Parcours voisins lignes
119  if(grid_prev->cells[i][j] == CELL_ALIVE){
120  cell_alive_count ++;
121  }
122  }
123  }
124  }
125  }
126  if((grid_prev->cells[x][y] == CELL_ALIVE && cell_alive_count == 2) || (cell_alive_count == 3)){ //Etat suivant de la cellule
127  grid_next->cells[x][y] = CELL_ALIVE;
128  }
129  else{
130  grid_next->cells[x][y] = CELL_DEAD;
131  }
132  }
133  }
134 }
